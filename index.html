💡 For full cursor-tracking, add the optional JS snippet from the README. Without JS, these use a centered "look-at" parallax on hover.

<div class="clover-magnetic-container"> <h3>Hover Me</h3> <p>Moves with your cursor.</p> </div> <!-- Optional JS for cursor tracking --> <script> document.querySelectorAll('.clover-magnetic-container').forEach(el => { el.addEventListener('mousemove', (e) => { const rect = el.getBoundingClientRect(); el.style.setProperty('--_mouse-x', (e.clientX - rect.left) / rect.width); el.style.setProperty('--_mouse-y', (e.clientY - rect.top) / rect.height); }); el.addEventListener('mouseleave', () => { el.style.setProperty('--_mouse-x', 0.5); el.style.setProperty('--_mouse-y', 0.5); }); }); </script>
